#include <stdio.h>
#define MAX 200
#define ID_LEN 11

void nhapMang(int kichThuoc, char mang[MAX][ID_LEN]) {
    for (int i = 0; i < kichThuoc; i++) {
        for (int j = 0; j < ID_LEN - 1; j++) {
            scanf("%c", &mang[i][j]);
        }
        mang[i][ID_LEN - 1] = '\0';
        getchar(); // bỏ \n
    }
}

void xuatMang(int kichThuoc, char mang[MAX][ID_LEN]) {
    for (int i = 0; i < kichThuoc; i++) {
        printf("%s\n", mang[i]);
    }
}

void tachSTT(char chuoi[], int *stt) {
    *stt = (chuoi[0] - '0') * 10 + (chuoi[1] - '0');
}

void copyChuoi(char dich[], char nguon[]) {
    int i = 0;
    while ((dich[i] = nguon[i]) != '\0') i++;
}

void gop2Chuoi(char mang[MAX][ID_LEN], int *soBanh,
               char banhCanChen[MAX][ID_LEN], int soBanhCanChen,
               int soID[], char tatCaBanh[MAX][ID_LEN]) {
    int len = 0;

    // Copy bánh gốc
    for (int i = 0; i < *soBanh; i++) {
        tachSTT(mang[i], &soID[len]);
        copyChuoi(tatCaBanh[len], mang[i]);
        len++;
    }

    // Copy bánh cần chèn
    for (int i = 0; i < soBanhCanChen; i++) {
        tachSTT(banhCanChen[i], &soID[len]);
        copyChuoi(tatCaBanh[len], banhCanChen[i]);
        len++;
    }

    *soBanh = len;

    // Copy tạm thời vào mảng chính
    for (int i = 0; i < len; i++) {
        copyChuoi(mang[i], tatCaBanh[i]);
    }
}

void sapXep(int soBanh, int soID[], char mang[MAX][ID_LEN]) {
    for (int i = 0; i < soBanh - 1; i++) {
        for (int j = i + 1; j < soBanh; j++) {
            if (soID[i] > soID[j]) {
                int tmp = soID[i];
                soID[i] = soID[j];
                soID[j] = tmp;

                char temp[ID_LEN];
                copyChuoi(temp, mang[i]);
                copyChuoi(mang[i], mang[j]);
                copyChuoi(mang[j], temp);
            }
        }
    }
}

int main() {
    char mang[MAX][ID_LEN];
    int soBanh;
    int soBanhCanChen;
    char banhCanChen[MAX][ID_LEN];
    char tatCaBanh[MAX][ID_LEN];
    int soID[MAX];

    printf("Nhap so banh ban dau:\n");
    scanf("%d", &soBanh);
    getchar();

    printf("Nhap id cac banh:\n");
    nhapMang(soBanh, mang);

    printf("Nhap so banh can them:\n");
    scanf("%d", &soBanhCanChen);
    getchar();

    printf("Nhap id cac banh can chen:\n");
    nhapMang(soBanhCanChen, banhCanChen);

    gop2Chuoi(mang, &soBanh, banhCanChen, soBanhCanChen, soID, tatCaBanh);

    sapXep(soBanh, soID, mang);

    printf("\nMang sau khi gop va sap xep:\n");
    xuatMang(soBanh, mang);

    return 0;
}
